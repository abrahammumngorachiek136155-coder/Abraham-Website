1. Write a C++ program to find out the largest among 3 numbers (using a class).


2. Write a C++ program to find the sum of the first N natural numbers (using a class). 

3. Write a C++ program to find the factorial of a number (using a class).

1 . answer:
#include <iostream>

class LargestNumberFinder {
private:
    int num1, num2, num3;

public:
    // Constructor to initialize the numbers
    LargestNumberFinder(int n1, int n2, int n3) {
        num1 = n1;
        num2 = n2;
        num3 = n3;
    }

    // Member function to find the largest number
    int findLargest() {
        if (num1 >= num2 && num1 >= num3) { //
            return num1;
        } else if (num2 >= num1 && num2 >= num3) { //
            return num2;
        } else {
            return num3; //
        }
    }
};

int main() {
    int n1, n2, n3;

    std::cout << "Enter three numbers: ";
    std::cin >> n1 >> n2 >> n3; //

    // Create an object of the LargestNumberFinder class
    LargestNumberFinder finder(n1, n2, n3);

    // Call the member function to find the largest number
    int largest = finder.findLargest();

    std::cout << "The largest number is: " << largest << std::endl; //

    return 0;
